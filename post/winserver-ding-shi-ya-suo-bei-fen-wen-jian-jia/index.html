winserver定时压缩备份文件夹

2019-12-09

#! /usr/bin/python
# version 1.1

import sys
import os
import time
import datetime

# 要备份的文件夹目录
fileFolder = r'D:\BPM'

# 要输出的文件名称(默认压缩在脚本执行的文件夹下)
# 需要自己在服务器上创建好fileName的目录
fileName = r'D:\Backup\BPM_bak_' + time.strftime('%Y%m%d%H') + '.zip'


exeCode = 'WinRAR a -r  %s %s' % (fileName, fileFolder)

# winrar a -r d:\scon.zip c:\scon  将c盘scon文件夹下的所有文件生成压缩并移动到d盘的目录下


def work():
    print('----开始备份----')
    os.system(exeCode)
    print('----备份完成----' + time.strftime('%Y%m%d%H'))


def main(hour, minute):
    print('工具启动')
    while True:
        while True:
            now = datetime.datetime.now()
            if now.hour == hour and now.minute == minute:
                break
        work()
        time.sleep(180)


main(1, 1)
